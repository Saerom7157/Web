CREATE TABLE user( 
    pid INT(11) NOT NULL AUTO_INCREMENT, //pid 필드 생성 숫자 1씩 증가
    name VARCHAR(100) NOT NULL,  //name 필드 생성 문자
    id VARCHAR(100) NOT NULL, //id 필드 생성 문자
    pw VARCHAR(100) NOT NULL, //pw 필드 생성 문자
    email VARCHAR(100) NOT NULL, //eamil 필드 생성 문자
    created DATETIME NOT NULL, //created 필드 생성 날짜
    status INT(1) NOT NULL,  //status 필드 생성 숫자
    PRIMARY KEY(pid)); //pid를 프라이머리키로 생성
